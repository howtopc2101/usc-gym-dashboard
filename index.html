async function loadData() {

    try {

        const res = await fetch("/api/log");

        const data = await res.json();


        document.getElementById("loading").style.display = "none";

        document.getElementById("content").style.display = "block";


        if (!data || !data.latest_counts) return;


        const c = data.latest_counts;


        const map = {

            village_strength: "Village Strength",

            village_cardio: "Village Cardio",

            lyon_weight: "Lyon Weight Room",

            lyon_cardio: "Lyon Cardio"

        };


        for (let id in map) {

            document.getElementById(id).textContent = c[id] ?? "--";

            document.getElementById(id + "_status").textContent =

                "Updated " + data.ran_at;

        }


    } catch (e) {

        document.getElementById("loading").textContent =

            "Failed to load data.";

    }

}


loadData();
